/* ─────────────────────────────────────────────
   Exemplo equivalente em C
   Cobre: tipos, arrays, funções, void,
   if/else, while, for, do-while, break, continue,
   return, scanf/printf, recursão, operadores.
   ───────────────────────────────────────────── */

#include <stdio.h>
#include <string.h>

/* ── Funções auxiliares ────────────────────────── */

int fatorial(int n)
{
    if (n <= 1) {
        return 1;
    }
    return n * fatorial(n - 1);
}

int potencia(int base, int exp)
{
    int resultado;
    resultado = 1;
    while (exp > 0) {
        resultado = resultado * base;
        exp = exp - 1;
    }
    return resultado;
}

int maximo(int a, int b)
{
    if (a > b) {
        return a;
    } else {
        return b;
    }
}

void bubbleSort(int arr[], int n)
{
    int i;
    int j;
    int temp;
    for (i = 0; i < n - 1; i = i + 1) {
        for (j = 0; j < n - i - 1; j = j + 1) {
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int somatorio(int arr[], int tam)
{
    int total;
    int i;
    total = 0;
    i = 0;
    while (i < tam) {
        total = total + arr[i];
        i = i + 1;
    }
    return total;
}

/* ── Programa principal ────────────────────────── */

int main(void)
{
    /* --- Tipos básicos --- */
    int a;
    int b;
    float pi;
    char letra;
    char mensagem[64];

    int contador;
    int i;
    int soma;
    int x;
    int v[8];
    int bin;
    int hex;
    char nome[64];
    int entrada;

    a = 10;
    b = 3;
    pi = 3.14159f;
    letra = 'A';
    strcpy(mensagem, "Ola, compilador!");

    puts(mensagem);

    /* --- Operadores aritméticos --- */
    printf("%d\n", a + b);
    printf("%d\n", a - b);
    printf("%d\n", a * b);
    printf("%d\n", a / b);
    printf("%d\n", a % b);

    /* --- Operadores relacionais e lógicos --- */
    if (a > b && b > 0) {
        puts("a maior que b, b positivo");
    }

    if (a == 10 || b == 0) {
        puts("a eh 10 ou b eh 0");
    }

    if (!(a < 0)) {
        puts("a nao eh negativo");
    }

    /* --- Operadores bit a bit --- */
    printf("%d\n", a & b);
    printf("%d\n", a | b);
    printf("%d\n", a ^ b);
    printf("%d\n", a << 2);
    printf("%d\n", a >> 1);
    printf("%d\n", ~a);

    /* --- if / else if / else --- */
    if (a > 100) {
        puts("grande");
    } else {
        if (a > 50) {
            puts("medio");
        } else {
            puts("pequeno");
        }
    }

    /* --- while com break --- */
    contador = 0;
    while (contador < 20) {
        if (contador == 7) {
            break;
        }
        contador = contador + 1;
    }
    printf("%d\n", contador);

    /* --- for com continue --- */
    soma = 0;
    for (i = 1; i <= 10; i = i + 1) {
        if (i % 2 == 0) {
            continue;
        }
        soma = soma + i;
    }
    printf("%d\n", soma);

    /* --- do-while --- */
    x = 1;
    do {
        x = x * 2;
    } while (x < 100);
    printf("%d\n", x);

    /* --- Arrays --- */
    for (i = 0; i < 8; i = i + 1) {
        v[i] = (i + 1) * (i + 1);
    }

    for (i = 0; i < 8; i = i + 1) {
        printf("%d\n", v[i]);
    }

    bubbleSort(v, 8);
    puts("Apos ordenacao:");
    for (i = 0; i < 8; i = i + 1) {
        printf("%d\n", v[i]);
    }

    puts("Somatório:");
    printf("%d\n", somatorio(v, 8));

    /* --- Recursão --- */
    puts("Fatorial de 10:");
    printf("%d\n", fatorial(10));

    puts("2 elevado a 8:");
    printf("%d\n", potencia(2, 8));

    puts("Maximo entre 42 e 17:");
    printf("%d\n", maximo(42, 17));

    /* --- Literais binário e hexadecimal --- */
    /* 0b1010 em C23; em C89/C99 use decimal ou expressão */
    bin = 10; /* equivalente a 0b1010 */
    hex = 0xFF;
    printf("%d\n", bin);
    printf("%d\n", hex);
    printf("%d\n", bin & hex);
    printf("%d\n", bin | hex);

    /* --- Strings e chars --- */
    strcpy(nome, "Compilador");
    puts(nome);
    printf("%c\n", letra);

    /* --- Leitura de entrada --- */
    puts("Digite um numero:");
    if (scanf("%d", &entrada) != 1) {
        return 1;
    }
    printf("%d\n", entrada);
    printf("%d\n", entrada * entrada);

    (void)pi; /* evita aviso se não usar pi em printf */
    return 0;
}
